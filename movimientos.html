import cv2
imagem = cv2.imread("gatito.jpg")
imagem_copy = imagem.copy()
angulo = 0

while True:
    cv2.imshow("gatito", imagem_copy)
    tecla = cv2.waitKey(0)
    if tecla == ord('w') or tecla == ord('W'):
        angulo += 90
    elif tecla == ord('a') or tecla == ord('A'):
        angulo += 80
    elif tecla == ord('s') or tecla == ord('S'):
        angulo += 60
    elif tecla == ord('d') or tecla == ord('D'):
        angulo += 180
    elif tecla == ord('x') or tecla == ord('X'):
        break
    elif tecla == 27:  # ESC
        break
    (h, w) = imagem.shape[:2]
    centro = (w // 2, h // 2)
    c_rotacion = cv2.getRotationMatrix2D(centro, angulo, 1.0)
    imagem_copy = cv2.warpAffine(imagem, c_rotacion, (w, h))

cv2.destroyAllWindows()
